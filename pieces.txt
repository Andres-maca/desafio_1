#include "pieces.h"
#include <random>
using  namespace std;

void piece_get_initial(PieceType type, unsigned short shape[4], int* width, int* height) {
    // Inicializar shape con ceros
    for (int i = 0; i < 4; i++) shape[i] = 0;
    
    switch (type) {
        case PIECE_I:
            shape[0] = 0b1111000000000000; // 4 bits en la primera fila
            *width = 4;
            *height = 1;
            break;
        case PIECE_O:
            shape[0] = 0b1100000000000000;
            shape[1] = 0b1100000000000000;
            *width = 2;
            *height = 2;
            break;
        case PIECE_T:
            shape[0] = 0b1110000000000000;
            shape[1] = 0b0100000000000000;
            *width = 3;
            *height = 2;
            break;
        case PIECE_S:
            shape[0] = 0b0110000000000000;
            shape[1] = 0b1100000000000000;
            *width = 3;
            *height = 2;
            break;
        case PIECE_Z:
            shape[0] = 0b1100000000000000;
            shape[1] = 0b0110000000000000;
            *width = 3;
            *height = 2;
            break;
        case PIECE_J:
            shape[0] = 0b1000000000000000;
            shape[1] = 0b1110000000000000;
            *width = 3;
            *height = 2;
            break;
        case PIECE_L:
            shape[0] = 0b0010000000000000;
            shape[1] = 0b1110000000000000;
            *width = 3;
            *height = 2;
            break;
        default:
            *width = 0;
            *height = 0;
            break;
    }
}

void piece_rotate(const unsigned short forma[4], int ancho, int alto,
                  unsigned short dar_forma[4], int* dar_ancho, int* dar_alto) {
    // Inicializar shape_out
    for (int i = 0; i < 4; i++) dar_forma[i] = 0;
    
    // Crear matriz de bits temporal
    int bits[4][4] = {{0}}; // maximo 4x4
    
    for (int r = 0; r < alto; r++) {
        unsigned short fila = forma[r];
        for (int c = 0; c < ancho; c++) {
            if (fila & (1 << (15 - c))) {
                bits[r][c] = 1;
            }
        }
    }
    
    // Rotar 90 grados horario: nueva_fila = c, nueva_columna = height_in - 1 - r
    int nueva_h = ancho;
    int nuevo_p = alto;
    
    for (int r = 0; r < alto; r++) {
        for (int c = 0; c < ancho; c++) {
            if (bits[r][c]) {
                int nr = c;
                int nc = alto - 1 - r;
                dar_forma[nr] |= (1 << (15 - nc));
            }
        }
    }
    
    *dar_ancho = nuevo_p;
    *dar_alto = nueva_h;
}
